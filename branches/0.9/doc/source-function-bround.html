52: /**
53:  * Redondeo bancario
54:  * @staticvar real $dFuzz
55:  * @param type $dVal
56:  * @param type $iDec
57:  * @return type
58:  */
59: function bround($dVal, $iDec=2)
60: {
61:    // banker's style rounding or round-half-even
62:    // (round down when even number is left of 5, otherwise round up)
63:    // $dVal is value to round
64:    // $iDec specifies number of decimal places to retain
65:    static $dFuzz = 0.00001; // to deal with floating-point precision loss
66:    $iRoundup = 0; // amount to round up by
67:    
68:    $iSign = ($dVal != 0.0) ? intval($dVal / abs($dVal)) : 1;
69:    $dVal = abs($dVal);
70:    
71:    // get decimal digit in question and amount to right of it as a fraction
72:    $dWorking = $dVal*pow(10.0,$iDec+1)-floor($dVal*pow(10.0,$iDec))*10.0;
73:    $iEvenOddDigit = floor($dVal*pow(10.0,$iDec))-floor($dVal*pow(10.0,$iDec-1))*10.0;
74:    
75:    if( abs($dWorking - 5.0) < $dFuzz )
76:       $iRoundup = ($iEvenOddDigit & 1) ? 1 : 0;
77:    else
78:       $iRoundup = ($dWorking>5.0) ? 1 : 0;
79:    
80:    return $iSign*((floor($dVal*pow(10.0,$iDec))+$iRoundup)/pow(10.0,$iDec));
81: }
